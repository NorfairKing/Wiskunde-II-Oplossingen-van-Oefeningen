\]
\[
\Leftrightarrow
\frac{Ax^2-Ab^2+Bx^2+Bxb+Cx^2-Cxb}{x(x^2-b^2)}
\]
\[
\Leftrightarrow

-Ab^2=1
\end{array}
\right.
\]
\[
\Leftrightarrow
\left\lbrace
\begin{array}{c c}

B = \frac{1}{2b^2}\\

\end{array}
\right.
\]
\[
\Leftrightarrow
\frac{1}{x(x^2-b^2} = -\frac{1}{b^2x^2} + \frac{1}{2b^2(x+b)} + \frac{1}{2b^2(x-b)}
